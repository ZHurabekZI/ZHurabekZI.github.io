InputData readInput(const string& filename) {
    ifstream file(filename);
    InputData data;
    map string, double* params = {
        {"b", &data.b}, {"a0", &data.a0}, {"l", &data.l}, {"L", &data.L},
        {"lam", &data.lam}, {"dP", &data.dP}, {"mu", &data.mu}, {"Q", &data.Q},
        {"k1", &data.k[0]}, {"k2", &data.k[1]}, {"k3", &data.k[2]},
        {"h1", &data.h[0]}, {"h2", &data.h[1]}, {"h3", &data.h[2]}
    };

    if (!file.is_open()) {
        cerr << "–û—à–∏–±–∫–∞: –ù–µ —É–¥–∞–ª–æ—Å—å –æ—Ç–∫—Ä—ã—Ç—å —Ñ–∞–π–ª " << filename << endl;
        exit(1);
    }

    string line;
    while (getline(file, line)) {
        size_t eqPos = line.find('=');
        if (eqPos == string::npos) continue;

        string key = line.substr(0, eqPos);
        double value = stod(line.substr(eqPos + 1));

        if (params.count(key)) {
            *params[key] = value;
        }
    }
    file.close();
    return data;
}

void Us(double *V, double *g, double *U, double *k, double *h, int n, double H, double t, 
        double lam, double L, double Q, double b, double a0, double dP, double mu) {
    double LA = lam * L;
    double T0 = (Q * t) / (2 * L * H * l);
    for (int i = 0; i < n; i++) {
        V[i] = (k[i] * H) / (k[0] * h[0] + k[1] * h[1] + k[2] * h[2]);
        g[i] = 1 + b * V[i] * a0 * T0 * (1 - exp(-LA));
        U[i] = -(k[i] * dP) / (mu * g[i] * L);
    }
}
